 36: /**
 37:  * Edge index.
 38:  *
 39:  * This class implements a graph edge node index, it can be used to replicate a Neo4j edge
 40:  * node to a MongoDB collection.
 41:  *
 42:  * Neo4j uses Lucene as its standard indexing tool, in this library we use MongoDB as the
 43:  * standard database, so it is a logical choice to use Mongo as the indexing mechanism for
 44:  * Neo4j. Also, by storing nodes in Mongo, by dumping the database contents we also have the
 45:  * graph structure with it.
 46:  *
 47:  * The class features a single data member, the {@link Node() node} which contains a Neo4j
 48:  * edge node reference. When {@link __construct() instantiating} this class you are required to
 49:  * provide a persistent instance of a node.
 50:  *
 51:  * The contents of the object will be the node properties as will be stored in the Mongo
 52:  * {@link kDEFAULT_CNT_EDGES default} collection for edges:
 53:  *
 54:  * <ul>
 55:  *  <li><i>{@link kTAG_LID kTAG_LID}</i>: This offset will hold the edge node ID.
 56:  *  <li><i>{@link kTAG_PATH kTAG_PATH}</i>: This offset will hold the edge node path, that
 57:  *      is, the source node ID, the predicate term {@link kTAG_GID identifier} and the
 58:  *      object node ID, all three separated by the
 59:  *      {@link kTOKEN_INDEX_SEPARATOR kTOKEN_INDEX_SEPARATOR} token.
 60:  *  <li><i>{@link kTAG_SUBJECT kTAG_SUBJECT}</i>: This offset will hold the subject node
 61:  *      information:
 62:  *   <ul>
 63:  *      <li><i>{@link kTAG_TERM kTAG_TERM}</i>: The subject node term
 64:  *          {@link kTAG_GID identifier}.
 65:  *      <li><i>{@link kTAG_NODE kTAG_NODE}</i>: The subject node ID.
 66:  *   </ul>
 67:  *  <li><i>{@link kTAG_PREDICATE kTAG_PREDICATE}</i>: This offset will hold the predicate
 68:  *      information:
 69:  *   <ul>
 70:  *      <li><i>{@link kTAG_TERM kTAG_TERM}</i>: The predicate term
 71:  *          {@link kTAG_GID identifier}, which corresponds to the edge node type.
 72:  *      <li><i>{@link kTAG_NODE kTAG_NODE}</i>: The edge node ID.
 73:  *   </ul>
 74:  *  <li><i>{@link kTAG_OBJECT kTAG_OBJECT}</i>: This offset will hold the object node
 75:  *      information:
 76:  *   <ul>
 77:  *      <li><i>{@link kTAG_TERM kTAG_TERM}</i>: The object node term
 78:  *          {@link kTAG_GID identifier}.
 79:  *      <li><i>{@link kTAG_NODE kTAG_NODE}</i>: The object node ID.
 80:  *   </ul>
 81:  *  <li><i>{@link kTAG_DATA kTAG_DATA}</i>: This offset will hold the edge node properties.
 82:  * </ul>
 83:  *
 84:  * The class provides a single member accessor method: {@link Node() Node}, which can be
 85:  * used to set the referenced edge node.
 86:  *
 87:  * <i>Note that the class will not cast to an array correctly, you must use the
 88:  * {@link getArrayCopy() getArrayCopy} method to get an array, if you know how to solve
 89:  * this, please do it!</i>
 90:  *
 91:  *  @package    MyWrapper
 92:  *  @subpackage Persistence
 93:  */

158: 
159:     /**
160:      * Build an edge node path.
161:      *
162:      * The edge path represents the unique identifier of an edge node using the relationship
163:      * node identifiers, it is composed by the subject and object node IDs and the predicate
164:      * term identifier.
165:      *
166:      * This method will return the {@link kTAG_PATH path} according to the provided
167:      * parameters:
168:      *
169:      * <ul>
170:      *  <li><b>$theSubject</b>: The relation subject node reference:
171:      *   <ul>
172:      *      <li><i>{@link COntologyNode COntologyNode}</i>: The node ID will be used.
173:      *      <li><i>Neo4j node</i>: The node ID will be used.
174:      *      <li><i>other</i>: Other types will be cast to a string.
175:      *   </ul>
176:      *  <li><b>$thePredicate</b>: The relation predicate term reference, the value will be
177:      *      cast to a string and is expected to represent the {@link COntologyTerm term}
178:      *      {@link kTAG_GID global} identifier.
179:      *  <li><b>$theObject</b>: The relation object node reference:
180:      *   <ul>
181:      *      <li><i>{@link COntologyNode COntologyNode}</i>: The node ID will be used.
182:      *      <li><i>Neo4j node</i>: The node ID will be used.
183:      *      <li><i>other</i>: Other types will be cast to a string.
184:      *   </ul>
185:      * </ul>
186:      *
187:      * The method will check if all three parameters are not empty, if this is not the case,
188:      * the method will raise an exception.
189:      *
190:      * @param mixed                 $theSubject         Subject node reference.
191:      * @param mixed                 $thePredicate       Predicate term reference.
192:      * @param mixed                 $theObject          Object node reference.
193:      *
194:      * @static
195:      * @return string
196:      */
197:     static function EdgeNodePath( $theSubject, $thePredicate, $theObject )
198:     {
199:         //
200:         // Check parameters.
201:         //
202:         $terms = array( 'Subject' => $theSubject,
203:                         'Predicate' => $thePredicate,
204:                         'Object' => $theObject );
205:         foreach( $terms as $key => $value )
206:         {
207:             //
208:             // Init node reference.
209:             //
210:             $node = NULL;
211:             
212:             //
213:             // Get node reference.
214:             //
215:             if( $value instanceof COntologyNode )
216:                 $node = $value->Node();
217:             elseif( $value instanceof Everyman\Neo4j\PropertyContainer )
218:                 $node = $value;
219:             
220:             //
221:             // Handle node.
222:             //
223:             if( $node !== NULL )
224:             {
225:                 if( $node->hasId() )
226:                     $terms[ $key ] = $node->getId();
227:                 else
228:                     throw new CException
229:                             ( "The $key node is not persistent",
230:                               kERROR_INVALID_STATE,
231:                               kMESSAGE_TYPE_ERROR,
232:                               array( $key => $value ) );                        // !@! ==>
233:             }
234:             
235:             //
236:             // Handle string.
237:             //
238:             else
239:             {
240:                 $string = (string) $value;
241:                 if( strlen( $string ) )
242:                     $terms[ $key ] = $string;
243:                 else
244:                     throw new CException
245:                             ( "The $key reference is empty",
246:                               kERROR_INVALID_PARAMETER,
247:                               kMESSAGE_TYPE_ERROR,
248:                               array( $key => $value ) );                        // !@! ==>
249:             }
250:         }
251:         
252:         return implode( kTOKEN_INDEX_SEPARATOR, $terms );                           // ==>
253: 
254:     } // EdgeNodePath.

258:      *  EdgeTermPath                                                                    *

261:     /**
262:      * Build an edge term path.
263:      *
264:      * This edge path represents an identifier of an edge node using the relationship term
265:      * identifiers, it is composed by the subject, predicate and object term
266:      * {@link kTAG_GID identifiers}.
267:      *
268:      * This method will return the {@link kTAG_EDGE_TERM path} according to the provided
269:      * parameters:
270:      *
271:      * <ul>
272:      *  <li><b>$theSubject</b>: The relation subject term reference, the value will be
273:      *      cast to a string and is expected to represent the {@link COntologyTerm term}
274:      *      {@link kTAG_GID global} identifier.
275:      *  <li><b>$thePredicate</b>: The relation predicate term reference, the value will be
276:      *      cast to a string and is expected to represent the {@link COntologyTerm term}
277:      *      {@link kTAG_GID global} identifier.
278:      *  <li><b>$theObject</b>: The relation object node reference, the value will be
279:      *      cast to a string and is expected to represent the {@link COntologyTerm term}
280:      *      {@link kTAG_GID global} identifier.
281:      * </ul>
282:      *
283:      * The method will check if all three parameters are not empty, if this is not the case,
284:      * the method will raise an exception.
285:      *
286:      * @param mixed                 $theSubject         Subject term reference.
287:      * @param mixed                 $thePredicate       Predicate term reference.
288:      * @param mixed                 $theObject          Object term reference.
289:      *
290:      * @static
291:      * @return string
292:      */
293:     static function EdgeTermPath( $theSubject, $thePredicate, $theObject )
294:     {
295:         //
296:         // Check parameters.
297:         //
298:         $terms = array( 'Subject' => $theSubject,
299:                         'Predicate' => $thePredicate,
300:                         'Object' => $theObject );
301:         foreach( $terms as $key => $value )
302:         {
303:             //
304:             // Cast to string.
305:             //
306:             $string = ( $value instanceof COntologyTerm )
307:                     ? $value->GID()
308:                     : (string) $value;
309:             
310:             //
311:             // Check if not empty.
312:             //
313:             if( strlen( $string ) )
314:                 $terms[ $key ] = $string;
315:             else
316:                 throw new CException
317:                         ( "The $key reference is empty",
318:                           kERROR_INVALID_PARAMETER,
319:                           kMESSAGE_TYPE_ERROR,
320:                           array( $key => $value ) );                            // !@! ==>
321:         }
322:         
323:         return implode( kTOKEN_INDEX_SEPARATOR, $terms );                           // ==>
324: 
325:     } // EdgeTermPath.

372:     /**
373:      * Locate node in container.
374:      *
375:      * We {@link COntologyNodeIndex::_LocateNode() overload} this method to locate
376:      * relationships in place of nodes.
377:      *
378:      * We also override the rule of raising an exception on zero node IDs, because the first
379:      * relationship will have an ID of zero, here we check if the identifier is numeric.
380:      *
381:      * Finally, we handle arrays in the node parameter: in this case we assume the array
382:      * holds the subject node ID, the predicate term {@link kTAG_GID identifier} and the
383:      * object node ID, these elements will be used to select the edge node whose
384:      * {@link kTAG_PATH kTAG_PATH} offset corresponds to this combination. In this case the
385:      * provided container must be an array structured as follows:
386:      *
387:      * <ul>
388:      *  <li><i>{@link kTAG_NODE kTAG_NODE}</i>: The graph container.
389:      *  <li><i>{@link kTAG_TERM kTAG_TERM}</i>: The index container.
390:      * </ul>
391:      *
392:      * @param mixed                 $theNode            Graph node identifier.
393:      * @param mixed                 $theContainer       Graph container.
394:      *
395:      * @access protected
396:      * @return Everyman\Neo4j\Node|NULL
397:      *
398:      * @throws {@link CException CException}
399:      *
400:      * @uses _ResolveIndexContainer()
401:      */
402:     protected function _LocateNode( $theNode, $theContainer )
403:     {
404:         //
405:         // Check graph container type.
406:         //
407:         if( $theContainer instanceof Everyman\Neo4j\Client )
408:         {
409:             //
410:             // Check node.
411:             //
412:             if( is_numeric( $theNode ) )
413:                 return $theContainer->getRelationship( $theNode );                  // ==>
414:             
415:             throw new CException
416:                     ( "Invalid node identifier",
417:                       kERROR_INVALID_PARAMETER,
418:                       kMESSAGE_TYPE_ERROR,
419:                       array( 'Node' => $theNode ) );                            // !@! ==>
420:         
421:         } // Graph container.
422:         
423:         throw new CException
424:                 ( "Unsupported container type",
425:                   kERROR_UNSUPPORTED,
426:                   kMESSAGE_TYPE_ERROR,
427:                   array( 'Container' => $theContainer ) );                      // !@! ==>
428:     
429:     } // _LocateNode.

436:     /**
437:      * Copy node properties to object.
438:      *
439:      * We {@link COntologyNodeIndex::_LocateNode() overload} this method to load the edge
440:      * node properties along with the subject, predicate and object information as follows:
441:      *
442:      * <ul>
443:      *  <li><i>{@link kTAG_LID kTAG_LID}</i>: This offset will hold the edge node ID.
444:      *  <li><i>{@link kTAG_PATH kTAG_PATH}</i>: This offset will hold the edge node path,
445:      *      that is, the source node ID, the predicate term {@link kTAG_GID identifier} and
446:      *      the object node ID, all three separated by the
447:      *      {@link kTOKEN_INDEX_SEPARATOR kTOKEN_INDEX_SEPARATOR} token.
448:      *  <li><i>{@link kTAG_SUBJECT kTAG_SUBJECT}</i>: This offset will hold the subject node
449:      *      information:
450:      *   <ul>
451:      *      <li><i>{@link kTAG_TERM kTAG_TERM}</i>: The subject node term
452:      *          {@link kTAG_GID identifier}.
453:      *      <li><i>{@link kTAG_NODE kTAG_NODE}</i>: The subject node ID.
454:      *   </ul>
455:      *  <li><i>{@link kTAG_PREDICATE kTAG_PREDICATE}</i>: This offset will hold the
456:      *      predicate information:
457:      *   <ul>
458:      *      <li><i>{@link kTAG_TERM kTAG_TERM}</i>: The predicate term
459:      *          {@link kTAG_GID identifier}, which corresponds to the edge node type.
460:      *      <li><i>{@link kTAG_NODE kTAG_NODE}</i>: The edge node ID.
461:      *   </ul>
462:      *  <li><i>{@link kTAG_OBJECT kTAG_OBJECT}</i>: This offset will hold the object node
463:      *      information:
464:      *   <ul>
465:      *      <li><i>{@link kTAG_TERM kTAG_TERM}</i>: The object node term
466:      *          {@link kTAG_GID identifier}.
467:      *      <li><i>{@link kTAG_NODE kTAG_NODE}</i>: The object node ID.
468:      *   </ul>
469:      *  <li><i>{@link kTAG_DATA kTAG_DATA}</i>: This offset will hold the edge node
470:      *      properties.
471:      * </ul>
472:      *
473:      * @access protected
474:      *
475:      * @uses Node()
476:      */
477:     protected function _LoadNodeProperties()
478:     {
479:         //
480:         // Load edge node properties.
481:         //
482:         parent::_LoadNodeProperties();
483:         
484:         //
485:         // Load subject info.
486:         //
487:         $data = Array();
488:         $node = $this->Node()->getStartNode();
489:         $subject_node = $data[ kTAG_NODE ] = $node->getId();
490:         if( ($tmp = $node->getProperty( kTAG_TERM )) !== NULL )
491:             $data[ kTAG_TERM ] = $tmp;
492:         $this->offsetSet( kTAG_SUBJECT, $data );
493:         
494:         //
495:         // Load predicate info.
496:         //
497:         $data = Array();
498:         $data[ kTAG_NODE ] = $this->Node()->getId();
499:         $predicate_term = $data[ kTAG_TERM ] = $this->Node()->getType();
500:         $this->offsetSet( kTAG_PREDICATE, $data );
501:         
502:         //
503:         // Load object info.
504:         //
505:         $data = Array();
506:         $node = $this->Node()->getEndNode();
507:         $object_node = $data[ kTAG_NODE ] = $node->getId();
508:         if( ($tmp = $node->getProperty( kTAG_TERM )) !== NULL )
509:             $data[ kTAG_TERM ] = $tmp;
510:         $this->offsetSet( kTAG_OBJECT, $data );
511:         
512:         //
513:         // Load path.
514:         //
515:         $this->offsetSet( kTAG_PATH,
516:                           self::EdgeNodePath
517:                             ( $subject_node, $predicate_term, $object_node ) );
518:     
519:     } // _LoadNodeProperties.

522:     /*===================================================================================
523:      *  _UpdateRelationshipCounts                                                       *
524:      *==================================================================================*/
525: 
526:     /**
527:      * Update relationship counts.
528:      *
529:      * This method will increment or decrement the relationships count in the subject and
530:      * object {node indexes.
531:      *
532:      * The method accepts the following parameters:
533:      *
534:      * <ul>
535:      *  <li><b>$theContainer</b>: The index Mongo container, we use the
536:      *      {@link CMongoContainer::Database() database} element and force the
537:      *      {@link kDEFAULT_CNT_NODES default} node collection name, since it apparently
538:      *      gives this class personality disorder problems to use the parent version of the
539:      *      {@link _ResolveIndexContainer() _ResolveIndexContainer} method.
540:      *  <li><b>$theModifiers</b>: This parameter represents the commit operation options,
541:      *      these will have been passed by the {@link Commit() commit} method.
542:      * </ul>
543:      *
544:      * These counts are stored in the {@link COntologyNodeIndex indexes} in the following
545:      * structure:
546:      *
547:      * <ul>
548:      *  <li><i>{@link kTAG_IN kTAG_IN}</i>: This element represents incoming relationships,
549:      *      it is an array structured as follows:
550:      *   <ul>
551:      *      <li><i>Key</i>: The term {@link kTAG_GID identifier} of the relationship
552:      *          predicate.
553:      *      <li><i>Value</i>: The number of incoming relationships for the given predicate.
554:      *   </ul>
555:      *  <li><i>{@link kTAG_OUT kTAG_OUT}</i>: This element represents outgoing
556:      *      relationships, it is an array structured as follows:
557:      *   <ul>
558:      *      <li><i>Key</i>: The term {@link kTAG_GID identifier} of the relationship
559:      *          predicate.
560:      *      <li><i>Value</i>: The number of outgoing relationships for the given predicate.
561:      *   </ul>
562:      * </ul>
563:      *
564:      * @param MongoContainer        $theContainer       Mongo container.
565:      * @param bitfield              $theModifiers       Commit modifiers.
566:      *
567:      * @access protected
568:      */
569:     protected function _UpdateRelationshipCounts( $theContainer, $theModifiers )
570:     {
571:         //
572:         // Resolve container.
573:         //
574:         $container = $theContainer->Database()->selectCollection( kDEFAULT_CNT_NODES );
575: 
576:         //
577:         // Init local storage.
578:         //
579:         $options = array( 'safe' => TRUE );
580:         $subject_id = $this->Node()->getStartNode()->getId();
581:         $predicate = $this->Node()->getType();
582:         $object_id = $this->Node()->getEndNode()->getId();
583:         
584:         //
585:         // Load subject.
586:         //
587:         $subject = $container->findOne( array( kTAG_LID => $subject_id ) );
588:         if( ! $subject )
589:             throw new CException
590:                     ( "Relationship subject index is missing",
591:                       kERROR_INVALID_STATE,
592:                       kMESSAGE_TYPE_ERROR,
593:                       array( 'Subject' => $subject_id ) );                      // !@! ==>
594:         
595:         //
596:         // Load object.
597:         //
598:         $object = $container->findOne( array( kTAG_LID => $object_id ) );
599:         if( ! $object )
600:             throw new CException
601:                     ( "Relationship object index is missing",
602:                       kERROR_INVALID_STATE,
603:                       kMESSAGE_TYPE_ERROR,
604:                       array( 'Object' => $object_id ) );                        // !@! ==>
605:         
606:         //
