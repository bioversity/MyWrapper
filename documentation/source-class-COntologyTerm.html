  29: /**
  30:  * Generic term.
  31:  *
  32:  * This kind of {@link COntologyTermObject term} represents a generic ontology term.
  33:  *
  34:  * This class handles the {@link _IsInited() inited} status by checking the following values
  35:  * in its {@link Kind() kinds} list:
  36:  *
  37:  * <ul>
  38:  *  <li><i>Default</i>: By default the object must have its {@link kTAG_CODE code} set.
  39:  *  <li><i>{@link kTYPE_NAMESPACE kTYPE_NAMESPACE}</i>: It uses the default requirements.
  40:  *  <li><i>{@link kTYPE_ROOT kTYPE_ROOT}</i>: It requires the {@link kTAG_NAME name} to be
  41:  *      set.
  42:  *  <li><i>{@link kTYPE_PREDICATE kTYPE_PREDICATE}</i>: It requires the
  43:  *      {@link kTAG_NAME name}.
  44:  *  <li><i>{@link kTYPE_ATTRIBUTE kTYPE_ATTRIBUTE}</i>: It requires the
  45:  *      {@link kTAG_NAME name} and {@link kTAG_CARDINALITY cardinality}.
  46:  *  <li><i>{@link kTYPE_MEASURE kTYPE_MEASURE}</i>: It requires the {@link kTAG_NAME name}
  47:  *      and {@link kTAG_TYPE type}.
  48:  *  <li><i>{@link kTYPE_ENUMERATION kTYPE_ENUMERATION}</i>: It requires the
  49:  *      {@link kTAG_NAME name} and {@link kTAG_ENUM enumeration}.
  50:  * </ul>
  51:  *
  52:  * In this class we ensure that the term has both the {@link Code() code} and
  53:  * {@link Name() name} in order to have an {@link _IsInited() inited}
  54:  * {@link kFLAG_STATE_INITED status}.

 122:     /**
 123:      * Manage term namespace.
 124:      *
 125:      * We {@link CTerm::NS() overload} this method in order to normalise the namespace: it
 126:      * must be provided as a string, so if you provide an object, it must be derived from
 127:      * this class and this method will use the provided object's {@link _index() index}.
 128:      *
 129:      * In this class we do not support instances as namespaces.
 130:      *
 131:      * @param mixed                 $theValue           Value.
 132:      * @param boolean               $getOld             TRUE get old value.
 133:      *
 134:      * @access public
 135:      * @return string
 136:      *
 137:      * @see kTAG_NAMESPACE
 138:      */
 139:     public function NS( $theValue = NULL, $getOld = FALSE )
 140:     {
 141:         //
 142:         // Normalise value.
 143:         //
 144:         if( ($theValue !== NULL)
 145:          && ($theValue !== FALSE) )
 146:         {
 147:             //
 148:             // Handle object.
 149:             //
 150:             if( $theValue instanceof self )
 151:             {
 152:                 //
 153:                 // Check if it has a code.
 154:                 //
 155:                 if( $theValue->Code() !== NULL )
 156:                     $theValue = $theValue->_index();
 157:                 
 158:                 else
 159:                     throw new CException
 160:                         ( "Invalid namespace reference",
 161:                           kERROR_UNSUPPORTED,
 162:                           kMESSAGE_TYPE_ERROR,
 163:                           array( 'Reference' => $theValue ) );                  // !@! ==>
 164:             
 165:             } // Provided object.
 166:             
 167:             //
 168:             // Convert to string
 169:             //
 170:             else
 171:                 $theValue = (string) $theValue;
 172:         
 173:         } // Provided new value.
 174:         
 175:         return parent::NS( $theValue, $getOld );                                    // ==>
 176: 
 177:     } // NS.

 221:     /**
 222:      * Manage code.
 223:      *
 224:      * We {@link COntologyTermObject::Code() overload} this method to check the format of
 225:      * the code, we perform the following checks:
 226:      *
 227:      * <ul>
 228:      *  <li><i>Characters</i>: There are a number of characters that are forbidden in
 229:      *      codes, providing a code with such characters will trigger an exception:
 230:      *   <ul>
 231:      *      <li><i>at (@)</i>: No code may start with this character, this is reserved for
 232:      *          identifying annotation terms.
 233:      *   </ul>
 234:      * </ul>
 235:      *
 236:      * @param mixed                 $theValue           Value.
 237:      * @param boolean               $getOld             TRUE get old value.
 238:      *
 239:      * @access public
 240:      * @return string
 241:      */
 242:     public function Code( $theValue = NULL, $getOld = FALSE )
 243:     {
 244:         //
 245:         // Intercept assignements.
 246:         //
 247:         if( ($theValue !== NULL)
 248:          && ($theValue !== FALSE) )
 249:         {
 250:             //
 251:             // Check ats.
 252:             //
 253:             if( substr( (string) $theValue, 0, 1 ) == '@' )
 254:                 throw new CException
 255:                     ( "Term codes may not start with the (@) sign",
 256:                       kERROR_INVALID_PARAMETER,
 257:                       kMESSAGE_TYPE_ERROR,
 258:                       array( 'Code' => $theValue ) );                           // !@! ==>
 259:         }
 260:         
 261:         return parent::Code( $theValue, $getOld );                                  // ==>
 262: 
 263:     } // Code.
