 377:     /**
 378:      * Set or return the exception severity.
 379:      *
 380:      * When providing <i>$theValue</i>:
 381:      *
 382:      * <ul>
 383:      *  <li><i>NULL</i>: Retrieve the current value.
 384:      *  <li><i>FALSE</i>: Reset the value to <i>NULL</i>.
 385:      *  <li><i>Integer</i>: Set the member to the provided value:
 386:      *   <ul>
 387:      *      <li><i>{@link kMESSAGE_TYPE_IDLE kMESSAGE_TYPE_IDLE}</i>: Idle state.
 388:      *      <li><i>{@link kMESSAGE_TYPE_NOTICE kMESSAGE_TYPE_NOTICE}</i>: A notice is an
 389:      *          informative message that does not imply an error, nor a situation that
 390:      *          should be handled; it can be considered as statistical data.
 391:      *      <li><i>{@link kMESSAGE_TYPE_MESSAGE kMESSAGE_TYPE_MESSAGE}</i>: A message is an
 392:      *          informative message that is addressed to somebody, although it does not
 393:      *          imply an error or warning, it was issued to a receiving party.
 394:      *      <li><i>{@link kMESSAGE_TYPE_WARNING kMESSAGE_TYPE_WARNING}</i>: Warnings are
 395:      *          informative data that indicate a potential problem, although they do not
 396:      *          imply an error, they indicate a potential problem or an issue that should be
 397:      *          addressed at least at a later stage.
 398:      *      <li><i>{@link kMESSAGE_TYPE_ERROR kMESSAGE_TYPE_ERROR}</i>: Errors indicate that
 399:      *          something prevented an operation from being performed, this does not
 400:      *          necessarily mean that the whole process is halted, but that the results of
 401:      *          an operation will not be as expected.
 402:      *      <li><i>{@link kMESSAGE_TYPE_FATAL kMESSAGE_TYPE_FATAL}</i>: Fatal errors are
 403:      *          {@link kMESSAGE_TYPE_ERROR errors} that result in stopping the whole
 404:      *          process: in this case the error will prevent other operations from being
 405:      *          performed and the whole process should be halted.
 406:      *      <li><i>{@link kMESSAGE_TYPE_BUG kMESSAGE_TYPE_BUG}</i>: Bugs, as opposed to
 407:      *          {@link kMESSAGE_TYPE_ERROR errors}, result from internal causes independant
 408:      *          from external factors. A bug indicates that an operation will never execute
 409:      *          as stated, it does not necessarily mean that it is
 410:      *          {@link kMESSAGE_TYPE_FATAL fatal}, but rather that the behaviour of an
 411:      *          operation does not correspond to its declaration.
 412:      *   </ul>
 413:      * </ul>
 414:      *
 415:      * The above mentioned codes are integer based, each state is represented by an
 416:      * interval, we encourage using the above mentioned codes, but this is not enforced.
 417:      *
 418:      * When setting or resetting the value, the method will return it <i>after</i> any
 419:      * modification was made.
 420:      *
 421:      * @param NULL|FALSE|integer    $theValue       NULL, FALSE or exception severity.
 422:      *
 423:      * @access public
 424:      * @return mixed    Exception severity or type.
 425:      */

 427:     {
 428:         //
 429:         // Return value.
 430:         //
 431:         if( $theValue === NULL )
 432:             return $this->mSeverity;                                                // ==>
 433:         
 434:         //
 435:         // Set or reset value.
 436:         //
 437:         $this->mSeverity = ( $theValue === FALSE )
 438:                          ? NULL
 439:                          : $theValue;
 440:         
 441:         return $this->mSeverity;                                                    // ==>

 450:     /**
 451:      * Set or return an exception reference.
 452:      *
 453:      * This method can be used to set, retrieve and remove exception references.
 454:      * References are a key/value pair that provide additional information on the exception.
 455:      *
 456:      * <ul>
 457:      *  <li><b>$theIndex</b>: This represents the reference label or name, if you provide
 458:      *      the parameter, it will be used as an array index to either retrieve, set or
 459:      *      remove elements, depending on the next parameter's value. If you omit the
 460:      *      parameter or pass <i>NULL</i> the method will consider the whole list of
 461:      *      references.
 462:      *  <li><b>$theValue</b>: This represents the reference value:
 463:      *   <ul>
 464:      *      <li><i>NULL</i>: This value indicates that we want to retrieve a reference. If
 465:      *          <i>$theIndex</i> is provided, the method will return the entry matching it,
 466:      *          if there is no match, the method will return <i>NULL</i>. If
 467:      *          <i>$theIndex</i> is omitted or <i>NULL</i>, the method will return the full
 468:      *          references array.
 469:      *      <li><i>FALSE</i>: This value indicates that we want to delete a reference. If
 470:      *          <i>$theIndex</i> is provided, the method will delete the entry matching it
 471:      *          and return <i>NULL</i>. If <i>$theIndex</i> is omitted or <i>NULL</i>, the
 472:      *          method will delete all references and return an empty array.
 473:      *      <li><i>Any other type</i>: Other types of value are considered as a value to
 474:      *          replace or be added to the references list: if <i>$theIndex</i> is provided,
 475:      *          the method will add or replace the matching element in the list and return
 476:      *          the eventual value <i>after</i> it was modified. If <i>$theIndex</i> is
 477:      *          omitted or <i>NULL</i>, the operation depends on the type of value:
 478:      *       <ul>
 479:      *          <li><i>array</i>: The provided array will replace the current list and the
 480:      *              method will return the list <i>after</i> it was replaced.
 481:      *          <li><i>Any other type</i>: The provided value will be appended to the list
 482:      *              by using a numeric index and the method will always return the provided
 483:      *              value.
 484:      *       </ul>
 485:      *   </ul>
 486:      * </ul>
 487:      *
 488:      * @param mixed             $theIndex       Reference element index.
 489:      * @param mixed             $theValue       NULL, FALSE, reference value or list.
 490:      *
 491:      * @access public
 492:      * @return mixed    Exception reference.
 493:      */
 494:     public function Reference( $theIndex = NULL, $theValue = NULL )
 495:     {
 496:         //
 497:         // Retrieve value.
 498:         //
 499:         if( $theValue === NULL )
 500:         {
 501:             //
 502:             // Return the whole list.
 503:             //
 504:             if( $theIndex === NULL )
 505:                 return $this->mReferences;                                          // ==>
 506:             
 507:             //
 508:             // Return list element.
 509:             //
 510:             return ( array_key_exists( $theIndex, $this->mReferences ) )
 511:                    ? $this->mReferences[ $theIndex ]                                // ==>
 512:                    : NULL;                                                          // ==>
 513:         }
 514:         
 515:         //
 516:         // Reset value.
 517:         //
 518:         if( $theValue === FALSE )
 519:         {
 520:             //
 521:             // Reset list.
 522:             //
 523:             if( $theIndex === NULL )
 524:             {
 525:                 //
 526:                 // List.
 527:                 //
 528:                 $this->mReferences = Array();
 529:                 
 530:                 return Array();                                                     // ==>
 531:             }
 532:             
 533:             //
 534:             // Reset element.
 535:             //
 536:             if( array_key_exists( $theIndex, $this->mReferences ) )
 537:                 unset( $this->mReferences[ $theIndex ] );
 538: 
 539:             return NULL;                                                            // ==>
 540:         
 541:         } // Reset value.
 542:         
 543:         //
 544:         // Add or replace element.
 545:         //
 546:         if( $theIndex !== NULL )
 547:             $this->mReferences[ $theIndex ] = $theValue;
 548:         
 549:         //
 550:         // Replace list.
 551:         //
 552:         elseif( is_array( $theValue ) )
 553:             $this->mReferences = $theValue;
 554:         
 555:         //
 556:         // Append element.
 557:         //
 558:         else
 559:             $this->mReferences[] = $theValue;
 560:         
 561:         return $theValue;                                                           // ==>
 562:     
 563:     } // Reference.

 579:     /**
 580:      * Display an exception.
 581:      *
 582:      * This method can be used to display exceptions in a web browser page.
 583:      *
 584:      * @param Exception|CException  $theException   Exception object.
 585:      *
 586:      * @static
 587:      * @return string
 588:      *
 589:      * @uses Reference()
 590:      * @uses Argument()
 591:      * @uses Value()
 592:      */
 593:     static function AsHTML( Exception $theException )
 594:     {
 595:         //
 596:         // Open table.
 597:         //
 598:         $document = new DOMDocument( '1.0', 'UTF-8' );
 599:         $table = $document->createElement( 'table' );
 600:         $table = $document->appendChild( $table );
 601:         $table->setAttribute( 'border', '1' );
 602:         $table->setAttribute( 'cellspacing', '2' );
 603:         $table->setAttribute( 'cellpadding', '0' );
 604:         $table->setAttribute( 'bordercolor', '#000000' );
 605:         
 606:         //
 607:         // Traverse trace.
 608:         //
 609:         $first = TRUE;
 610:         $trace = self::_Trace( $theException );
 611:         while( count( $trace ) )
 612:         {
 613:             //
 614:             // Add divider.
 615:             //
 616:             if( ! $first )
 617:             {
 618:                 $row = $document->createElement( 'tr' );
 619:                 $row = $table->appendChild( $row );
 620:                 $col = $document->createElement( 'td' );
 621:                 $col = $row->appendChild( $col );
 622:                 $col->setAttribute( 'colspan', '3' );
 623:                 $col->setAttribute( 'height', '8' );
 624:             }
 625:             else
 626:                 $first = FALSE;
 627:             
 628:             //
 629:             // Get element.
 630:             //
 631:             $element = array_shift( $trace );
 632:             
 633:             //
 634:             // Populate table.
 635:             //
 636:             if( $element instanceof Exception )
 637:                 self::_Exception2HTML( $table, $element );
 638:             else
 639:                 self::_Trace2HTML( $table, $element );
 640:         
 641:         } // Traversing trace.
 642:         
 643:         return $document->saveHTML();                                               // ==>
 644:     
 645:     } // AsHTML.
